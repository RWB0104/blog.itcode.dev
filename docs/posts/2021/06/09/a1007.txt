			}
		}
	}

	{
		int x = 0;
		int y = 0;
		
		for (int i = 0; i < P.length; i++)
		{
			// ÏñëÏàòÎ°ú ÏÑ†ÌÉùÎêú Ï†êÏùº Í≤ΩÏö∞
			if (isChecked[i])

				y -= P[i][1];
			}
		}
		
		return Math.sqrt(Math.pow(x, 2) + Math.pow(y, 2));
	}
