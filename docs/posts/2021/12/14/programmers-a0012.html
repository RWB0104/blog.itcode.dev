class Solution
{
    private static final Queue<Integer> ONE = new LinkedList<>();
    private static final Queue<Integer> TWO = new LinkedList<>();
    private static final Queue<Integer> THREE = new LinkedList<>();
    
    /**
     * í•´ë‹µ ë°˜í™˜ ë©”ì„œë“œ
     *
     * @param answers: [int[]] ìµœê³  ë“ì ì
     *
     * @return [int[]] í•´ë‹µ
     */
    public int[] solution(int[] answers)
    {
        initQueue();
        
        int[] counts = { 0, 0, 0 };
        
        for (int item : answers)
        {
            int one = Objects.requireNonNull(ONE.poll());
            int two = Objects.requireNonNull(TWO.poll());
            int three = Objects.requireNonNull(THREE.poll());
            
            counts[0] += item == one ? 1 : 0;
            counts[1] += item == two ? 1 : 0;
            counts[2] += item == three ? 1 : 0;
            
            ONE.add(one);
            TWO.add(two);
            THREE.add(three);
        }
        
        int max = Arrays.stream(counts).max().getAsInt();
        
        ArrayList<Integer> list = new ArrayList<>();
        
        for (int i = 0; i < counts.length; i++)
        {
            // ìµœê³  ì ìˆ˜ë¥¼ ì–»ì—ˆì„ ê²½ìš°
            if (counts[i] == max)
            {
                list.add(i + 1);
            }
        }
        
        return list.stream().mapToInt(Integer::intValue).toArray();
    }
    
    /**
     * í ì´ˆê¸°í™” ë©”ì„œë“œ
     */
    private void initQueue()
    {
        ONE.clear();
        TWO.clear();
        THREE.clear();
        
        ONE.add(1);
        ONE.add(2);
        ONE.add(3);
        ONE.add(4);
        ONE.add(5);
        
        TWO.add(2);
        TWO.add(1);
        TWO.add(2);
        TWO.add(3);
        TWO.add(2);
        TWO.add(4);
        TWO.add(2);
        TWO.add(5);
        
        THREE.add(3);
        THREE.add(3);
        THREE.add(1);
        THREE.add(1);
        THREE.add(2);
        THREE.add(2);
        THREE.add(4);
        THREE.add(4);
        THREE.add(5);
        THREE.add(5);
    }
}
