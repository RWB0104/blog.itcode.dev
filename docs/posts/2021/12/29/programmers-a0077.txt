	public int[] solution(int rows, int columns, int[][] queries)
	{
		int[] answer = new int[queries.length];
		
		int[][] board = new int[rows][columns];
		
		for (int i = 0; i < rows; i++)
		{
			for (int j = 0; j < columns; j++)
